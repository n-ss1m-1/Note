CMake
注释：行注释#   块注释#[[]]

编写和使用CMake：
1. touch CMakeLists.txt
2. 表明cmake最低版本：cmake_minimum_required(VERSION 3.0)
3. 表明项目名称：project(CALC)
4. 生成可执行文件：add_executable(可执行程序名 源文件名1 源文件名2 ...)   #源文件间也可以用分号(;)分隔
5. cmake CmakeLists.txt文件所在路径  => 生成makefile等文件(到当前目录下)
6. make(在makefile所在目录下调用)  =>  生成可执行程序
7. 可以创建build目录 管理cmake系列文件

定义变量 =》set的使用：
语法([]为可选项)：SET(VAR [VALUE] [CACHE TYPE DOCSTRING [FORCE]])
例子：
    set(SRC_LIST add.cc div.cc main.cc)    #源文件也可以使用分号(;)分隔
    add_executable(app ${SRC_LIST})

指定使用的C++标准 =》set的使用：
方法1：在 CMakeLists.txt 中通过 set 命令指定
    set(CMAKE_CXX_STANDARD 11)            #-std=c++11
方法2：在执行 cmake 命令的时候指定出这个宏的值
    cmake CMakeLists.txt文件路径 -DCMAKE_CXX_STANDARD=11

指定可执行程序输出的路径(EXECUTABLE_OUTPUT_PATH) =》set的使用：
set(HOME /home/robin/Linux/Sort)
set(EXECUTABLE_OUTPUT_PATH ${HOME}/bin)
注：此处使用的是绝对路径  |||  如果使用相对路径，则./对应的就是makefile文件所在的目录

list的使用方法：
1.字符串拼接：list(APPEND <list> [<element> ...])
		list：表示已定义的变量
		element：表示某个字符串参数
2.字符串移除：list(REMOVE_ITEM <list> <element> [<element> ...])
	例子：list(REMOVE_ITEM SRC ${PROJECT_SOURCE_DIR}/main.cc)
3......

搜索源文件：
方法1：aux_source_directory(< dir > < var >) =》 搜索dir目录下的源文件，储存到var变量中
注：宏PROJECT_SOURCE_DIR和宏CMAKE_CURRENT_SOURCE_DIR  表示CMakeLists.txt所在目录
例：aux_source_directory(${CMAKE_CURRENT_SOURCE_DIR}/src SRC_LIST)

方法2：file(GLOB/GLOB_RECURSE 变量名 要搜索的文件路径和文件类型)
注：GLOB：搜索指定目录     GLOB_RECURSE：递归搜素指定目录
例：file(GLOB MAIN_HEAD ${CMAKE_CURRENT_SOURCE_DIR}/include/*.h)
若源文件分布在多个目录=》使用set合并变量：set(SRC SRC1 SRC2)

包含头文件：
include_directories(headpath)        #headpath为头文件所在目录
例：include_directories(${PROJECT_SOURCE_DIR}/include)

制作静态库：
add_library(库名字 STATIC 源文件1 [源文件2] ...)
(linux)静态库名字分为三部分：lib+库名字+.a 此处只需指定出库名字即可

制作动态库：
add_library(库名称 SHARED 源文件1 [源文件2] ...)
(linux)动态库名字分为三部分：lib+库名字+.so 此处只需指定出库名字即可

设置静态库/动态库的生成路径：
set(LIBRARY_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/lib)

链接静态库(应该将该命令写到生成可执行文件之前)：
link_directories(<lib path>)        #非系统提供的库->指定静态库所在目录
link_libraries(<静态库1名字> [<静态库名字2>...])

链接动态库(▲应该将该命令写到生成可执行文件之后)：
link_directories(<lib path>)        #非系统提供的库->指定动态库所在目录
target_link_libraries(
    <target>
    <PRIVATE|PUBLIC|INTERFACE> <item>...
    [<PRIVATE|PUBLIC|INTERFACE> <item>...]...)
    target：源文件/动态库文件/静态库文件/【可执行文件】
    PUBLIC：可传递到任意链接
    PRIVATE：不可传递到第三者
    INTERFACE：不可传递到第二者，只会导出符号(即函数说明)
    item：动态库名称

嵌套CMake：
指定子目录：add_subdirectory(source_dir [binary_dir] [EXCLUDE_FROM_ALL])
    source_dir：指定了CMakeLists.txt所在子目录
    binary_dir：指定了输出文件的路径

示例2：muduo网络库构建的例子（单目录）
cmake_minimum_required(VERSION 3.10)
project(muduo)
set(CMAKE_CXX_STANDARD 11)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/include)
file(GLOB_RECURSE SRC ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cc)
set(EXECUTABLE_OUTPUT_PATH ${CMAKE_CURRENT_SOURCE_DIR}/bin)
add_executable(main ${SRC})
target_compile_options(main PRIVATE -Wall -Wextra -g -O0)
target_link_libraries(main PRIVATE pthread)

示例3：链接静态库
cmake_minimum_required(VERSION 3.10)
project(muduo)
set(CMAKE_CXX_STANDARD 11)
#包含头文件
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/include)
#包含源文件
file(GLOB_RECURSE SRC ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cc)
list(REMOVE_ITEM SRC ${CMAKE_CURRENT_SOURCE_DIR}/src/main.cc)    #排除main.cc
#生成静态库+设置静态库生成路径
set(LIBRARY_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/lib)
add_library(src_lib STATIC ${SRC})
#指明第三方静态库的路径+链接静态库
link_directories(${PROJECT_SOURCE_DIR}/lib)
link_libraries(src_lib)
#设置可执行文件的生成路径
set(EXECUTABLE_OUTPUT_PATH ${CMAKE_CURRENT_SOURCE_DIR}/bin)
#生成可执行文件
add_executable(main ./src/main.cc)
target_compile_options(main PRIVATE -Wall -Wextra -g -O0)
target_link_libraries(main pthread)

示例5：嵌套CMake
#muduo/CMakeLists.txt
cmake_minimum_required(VERSION 3.10)			
project(muduo)
set(CMAKE_CXX_STANDARD 11)
#设置全局变量
set(LIB_PATH ${PROJECT_SOURCE_DIR}/lib)
set(EXEC_PATH ${PROJECT_SOURCE_DIR}/bin)
set(HEAD_PATH ${PROJECT_SOURCE_DIR}/include)
#设置名称
set(SRC_LIB_NAME src)
set(LOG_LIB_NAME log)
set(EXEC_NAME main)
#添加子目录
add_subdirectory(./src)
add_subdirectory(./log)

list(REMOVE_ITEM SRC ./main.cc)		#排除main.cc
set(LIBRARY_OUTPUT_PATH ${LIB_PATH})	#设置动态库生成路径
add_library(${SRC_LIB_NAME} SHARED ${SRC})	#生成动态库
set(EXECUTABLE_OUTPUT_PATH ${EXEC_PAHT})	#设置可执行程序生成路径
add_executable(${EXEC_NAME} ./main.cc)		#生成可执行程序
target_compile_options(main PRIVATE -Wall -Wextra -g -O0)	#编译选项
